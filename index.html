df %>% 
  filter(zona == 1, !is.na(identificador)) %>% 
  mutate(
    sexo = factor(sexo, levels = c(0, 1), labels = c("Female", "Male")),
    faixaetaria = factor(faixaetaria, 
                         levels = c(1, 2, 3, 4), 
                         labels = c("50-59", "60-69", "70-79", "≥80")),
    cor_raca = factor(cor_raca, levels = c(0, 1), labels = c("White", "Non-white")),
    escolaridade2 = factor(escolaridade2, 
                           levels = c(1, 2, 3), 
                           labels = c("<4", "4-7", "≥8")),
    est_civil = factor(est_civil, 
                       levels = c(0, 1), 
                       labels = c("Unmarried", "Married or cohabiting")),
    dcnt = factor(dcnt, levels = c(0, 1, 2), labels = c("0", "1", "≥2")),
    aas = factor(aas, levels = c(0, 1), 
                 labels = c("Very poor/Poor", "Excellent/Very good/Good/Fair")),
    rendadompc2corr = factor(rendadompc2corr, 
                             levels = c(1, 2, 3), 
                             labels = c("1st tertile: ≤933", 
                                        "2nd tertile: >933-1,500", 
                                        "3rd tertile: >1,500")),
    anos_municipio = as.numeric(anos_municipio),
    cair_passeios = factor(cair_passeios, levels = c(0,1), labels = c("No","Yes")),
    subir_transporte = factor(subir_transporte, levels = c(0,1), labels = c("No","Yes")),
    atravessar_rua = factor(atravessar_rua, levels = c(0,1), labels = c("No","Yes")),
    bom_lugar = factor(bom_lugar, levels = c(0,1), labels = c("No","Yes")),
    bom_lugar_brincar = factor(bom_lugar_brincar, levels = c(0,1), labels = c("No","Yes")),
    agradavel_exercicios = factor(agradavel_exercicios, levels = c(0,1), labels = c("No","Yes"))
  ) %>% 
  as_survey_design(ids = upa, strata = estrato, weights = peso_calibrado, nest = TRUE) %>%
  tbl_svysummary(
    by = sexo,
    include = c(
      faixaetaria, cor_raca, escolaridade2, est_civil, dcnt, aas,
      rendadompc2corr, anos_municipio,
      cair_passeios, subir_transporte, atravessar_rua,
      bom_lugar, bom_lugar_brincar, agradavel_exercicios
    ),
    type = list(
      cair_passeios ~ "categorical",
      subir_transporte ~ "categorical",
      atravessar_rua ~ "categorical",
      bom_lugar ~ "categorical",
      bom_lugar_brincar ~ "categorical",
      agradavel_exercicios ~ "categorical"
    ),
    label = list(
      faixaetaria ~ "Age group (years)",
      cor_raca ~ "Race/Color",
      escolaridade2 ~ "Education (years)",
      est_civil ~ "Marital status",
      dcnt ~ "Number of chronic diseases",
      aas ~ "Self-rated health",
      rendadompc2corr ~ "Household income per capita in tertiles (R$)",
      anos_municipio ~ "Length of residence (years) [mean (SD)]",
      cair_passeios ~ "Fear of falling due to sidewalk defects",
      subir_transporte ~ "Difficulty boarding public transport",
      atravessar_rua ~ "Difficulty crossing the street",
      bom_lugar ~ "Neighborhood is a good place to live",
      bom_lugar_brincar ~ "Good neighborhood for children and adolescents",
      agradavel_exercicios ~ "Neighborhood pleasant for walking, running, or cycling"
    ),
    statistic = list(
      all_categorical() ~ "{p}",
      all_continuous() ~ "{mean} ({sd})"
    ),
    digits = list(
      all_categorical() ~ 1,
      all_continuous() ~ 1
    ),
    missing = "no"
  ) %>%
  add_overall(last = FALSE) %>%
  add_p(
      pvalue_fun = ~ style_pvalue(.x, digits = 3)
      ) %>% 
  add_ci(
    style_fun = list(
      all_categorical() ~ function(x) sprintf("%.1f", x * 100),
      all_continuous() ~ function(x) style_sigfig(x, digits = 1)
    )
  ) %>% 
  modify_table_body(
    ~ .x %>%
      dplyr::mutate(
        dplyr::across(
          starts_with("ci_stat"),
          ~ ifelse(variable == "anos_municipio", "", .x)
        )
      )
  ) %>% 
  modify_fmt_fun(
    update = list(

      }
    )
  ) %>%
  modify_header(

  lbl_list[[exp_var]] <- exposure_labels[[exp_var]]

    model_fit_ct,
    include = all_of(exp_var),
    exponentiate = TRUE,

Multilevel poisson models for neighborhood indicators and resident perceptions (census tract nested within city) 
analysis_data_ct_city_poisson <- df %>%
  filter(zona == 1, !is.na(identificador)) %>%
  mutate(
    city = factor(cidade2),
    census_tract = factor(setor),
    normalized_weight = peso_calibrado / mean(peso_calibrado, na.rm = TRUE)
  )

run_model_row_ct_city_poisson <- function(outcome_var, exp_var) {

  model_formula <- as.formula(
    paste(outcome_var, "~", exp_var, "+",
          paste(adjustments, collapse = " + "),
          "+ (1 | city/census_tract)")
  )

  model_fit_ct_city_poisson <- glmmTMB(
    formula = model_formula,
    data = analysis_data_ct_city_poisson,
    family = poisson(link = "log"),
    weights = normalized_weight
  )

  lbl_list <- list()
  lbl_list[[exp_var]] <- exposure_labels[[exp_var]]

  tbl_regression(
    model_fit_ct_city_poisson,
    include = all_of(exp_var),
    exponentiate = TRUE,
    label = lbl_list,
    estimate_fun = function(x) style_number(x, digits = 1),
    pvalue_fun = function(x) style_pvalue(x, digits = 3),
    show_single_row = all_of(exp_var)
  ) %>%
    modify_table_body(
      ~ .x %>%
        mutate(ci = gsub(",", " -", ci)) %>%
        select(-conf.low, -conf.high)
    ) %>%
    modify_header(
      label = "**Neighborhood indicators**",
      estimate = "**PR**",
      ci = "**95% CI**"
    )
}

table_list_ct_city_poisson <- map(outcome_vars, function(out_v) {
  sub_tables <- map(exposure_vars, ~run_model_row_ct_city_poisson(out_v, .x))
  tbl_stack(sub_tables)
})

final_table_ct_city_poisson <- tbl_stack(
  tbls = table_list_ct_city_poisson,
  group_header = paste0("**", outcome_labels[outcome_vars], "**")
)
